 30 /**
 31  * @construct
 32  * @param egn String   
 33  */
 34 function EGN(egn) {
 35 	this.value = egn;
 36 
 37 	this._year = +egn.slice(0, 2); 
 38 	this._month = +egn.slice(2, 4); 
 39 	this._date = +egn.slice(4, 6); 
 40 	this._order = +egn.slice(6, 9);
 41 }
 42 
 43 EGN.prototype = {
 44 	constructor : EGN,
 45 	/**
 46 	  * Съдържа имената на всички региони в България, в които може да бъде записан човек при неговото раждане.
 47 	  * @type Array
 48 	  */
 49 	regionNames : [
 50 		'Благоевград',
 51 		'Бургас',
 52 		'Варна',
 53 		'Велико Търново',
 54 		'Видин',
 55 		'Враца',
 56 		'Габрово',
 57 		'Кърджали',
 58 		'Кюстендил',
 59 		'Ловеч',
 60 		'Монтана',
 61 		'Пазарджик',
 62 		'Перник',
 63 		'Плевен',
 64 		'Пловдив',
 65 		'Разград',
 66 		'Русе',
 67 		'Силистра',
 68 		'Сливен',
 69 		'Смолян',
 70 		'София - град',
 71 		'София - окръг',
 72 		'Стара Загора',
 73 		'Добрич (Толбухин)',
 74 		'Търговище',
 75 		'Хасково',
 76 		'Шумен',
 77 		'Ямбол',
 78 		'Друг/Неизвестен'
 79 	],
 80 	
 81 	/**
 82 	 * Съдържа отделените интервали за всеки един регион. 
 83 	 * Интервала е: (x, y]
 84 	 * Използва се за определяне на региона.
 85 	 * @private
 86 	 * @type Array
 87 	 */
 88 	_regionIds : [-1, 43, 93, 139, 169, 183, 217, 233, 281, 301, 319, 341, 377, 395, 435, 501, 527, 555, 575, 601, 623, 721, 751, 789, 821, 843, 871, 903, 925, 999],	
 89 	
 90 	/**
 91 	 * На базата на числата в масива се изчислява 10-тата контролна цифра в ЕГН.
 92 	 * Използва се за валидация на ЕГН.
 93 	 * @private
 94 	 * @type Array
 95 	 */
 96 	_weights : [2, 4, 8, 5, 10, 9, 7, 3, 6],
 97 	
 98 	/**
 99 	  * Проверява валидността на ЕГН.
100 	  * Датата на раждане се проверява за валидност спрямо Григорианският календар.
101 	  * @type Boolean value
102 	  */
103 	isValid : function() 
104 	{
105 		var egn = this.value,
106 			bornDate = this.getBornDate(),
107 			month = (this._month % 20) - 1;
108 			
109 		if (egn.length != 10 || /\D/.test(egn) || bornDate.getMonth() != month || bornDate.getDate() != this._date) 
110 		{
111 			return false;
112 		}
113 		
114 		egn = egn.split('');
115 		for (var i = this._weights.length, sum = 0; i--;)
116 		{
117 			sum += this._weights[i] * egn[i];
118 		}
119 		if (sum % 11 % 10 != egn[9]) 
120 		{
121 			return false;
122 		}
123 		
124 		return true;
125 	},
126 	
127 	/**
128 	  * Извлича датата на раждане.
129 	  * @type Date
130 	  */
131 	getBornDate : function() 
132 	{
133 		var year = 1900;
134 		if (this._month > 40) 
135 		{
136 			year += 100;
137 		}
138 		else if (this._month > 20) 
139 		{
140 			year -= 100;
141 		}
142 		
143 		return new Date(year + this._year, (this._month % 20) - 1, this._date);
144 	},
145 	
146 	/**
147 	  * Изчислява региона, на който отговаря ЕГН-то.
148 	  * @type Number value
149 	  * @return Името на региона може да се вземе като се използва върнатия резултат за индекс в масива <code>regionNames</code>.
150 	  * @see #regionNames
151 	  */
152 	getRegion : function() 
153 	{
154 		if (typeof this._region != 'number') 
155 		{
156 			var start = 0,
157 				end = this._regionIds.length - 1,
158 				middle, lower_b, upper_b;
159 			
160 			while(start <= end)
161 			{
162 				middle = Math.floor((end + start) / 2);
163 				lower_b = this._regionIds[middle];
164 				upper_b = this._regionIds[middle + 1];
165 				if (this._order > upper_b) 
166 				{	
167 					start = middle + 1;
168 				}
169 				else if (this._order < lower_b)
170 				{
171 					end = middle - 1;
172 				}
173 				else {
174 					this._region = middle;
175 					break;
176 				}
177 			}
178 		}
179 		return this._region;
180 	},
181 	
182 	/** 
183 	  * Изчислява поредността на раждане за региона.
184 	  * @type Number value
185 	  */
186 	getOrderByRegion : function() 
187 	{
188 		return this._order - this._regionIds[this.getRegion()];
189 	},
190 	
191 	/** 
192 	  * Изчислява поредността на раждане за региона, спрямо хората от същия пол.
193 	  * @type Number value
194 	  */	
195 	getSexOrderByRegion : function()
196 	{
197 		return Math.ceil(this.getOrderByRegion() / 2);
198 	},
199 	
200 	/**
201 	  * Определяне на пола.
202 	  * @type Number value
203 	  * @return 0 - мъж<br /> 1 - жена
204 	  */
205 	getSex : function() 
206 	{
207 		return this._order % 2;
208 	}
209 };
